//Revamped support table thingy
/*
00 - word - number of supports

04 - byte - character id 1
05 - byte - initial support level
06 - byte - 0 for no S/A+, 1 for A+, 2 for S
07 - byte - spacer
08 - byte - chapter id of C unlock
09 - byte - chapter id of B unlock
0A - byte - chapter id of A unlock
0B - byte - chapter id of A+/S unlock
0C - word - C support event pointer
10 - word - B support event pointer
14 - word - A support event pointer
18 - word - A+/S support event pointer

1C - byte - character id 2
etc

Support level:
0 = no support
1 = C support available
2 = C support done
3 = B support available
4 = B support done
5 = A support available
6 = A support done
7 = S/A+ support available
8 = S/A+ support done

Example:
SetSupportsPointer(Roy, RoySupports)
RoySupports:
WORD 2 //number of supports
SetSupportEntry(Lilina, 0, 2, 8, 16, 22, 27, RoyLilinaC, RoyLilinaB, RoyLilinaA, RoyLilinaS)
SetSupportEntry(Lalum, 0, 2, 9, 20, 23, 30, RoyLalumC, RoyLalumB, RoyLalumA, RoyLalumS)
*/

#define CharacterEditorTable 0x83FFC0

#define SetSupportsPointer(character_id, support_pointer) "PUSH; ORG CharacterEditorTable+(48*character_id)+44; POIN support_pointer; POP"

#define SetSupportEntry(support_char_id, initial_level, final_pairing, C_unlock_ch, B_unlock_ch, A_unlock_ch, S_unlock_ch, C_event_ptr, B_event_ptr, A_event_ptr, S_event_ptr) "BYTE support_char_id initial_level final_pairing 0 C_unlock_ch B_unlock_ch A_unlock_ch S_unlock_ch; POIN C_event_ptr B_event_ptr A_event_ptr S_event_ptr"
